    <script>
        // 1. Generate Floating Hearts
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart');
            heart.innerHTML = '❤️';
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.top = '100vh';
            heart.style.fontSize = (Math.random() * 20 + 10) + 'px';
            heart.style.opacity = Math.random();
            
            document.getElementById('canvas-container').appendChild(heart);

            gsap.to(heart, {
                y: -window.innerHeight - 100,
                x: `+=${(Math.random() - 0.5) * 200}`,
                duration: Math.random() * 3 + 4,
                ease: "power1.out",
                onComplete: () => heart.remove()
            });
        }

        // 2. Initial Entrance Animation
        const intro = document.getElementById('intro-overlay');
        const card = document.getElementById('birthday-card');

        intro.addEventListener('click', () => {
            // Hide intro
            gsap.to(intro, {
                opacity: 0, 
                duration: 1, 
                onComplete: () => intro.style.display = 'none'
            });

            // Show card
            gsap.to(card, {
                opacity: 1,
                scale: 1,
                duration: 1.5,
                delay: 0.5,
                ease: "back.out(1.7)"
            });

            // Start hearts
            setInterval(createHeart, 300);
        });

        // 3. The "AI" Secret Reveal
        function revealSecret() {
            const secret = document.getElementById('special-message');
            const btn = document.querySelector('.btn');
            
            gsap.to(btn, { scale: 0, duration: 0.3, onComplete: () => btn.style.display = 'none' });
            
            secret.style.display = 'block';
            gsap.from(secret, {
                y: 20,
                opacity: 0,
                duration: 1,
                ease: "power2.out"
            });

            // Extra explosion of hearts
            for(let i=0; i<15; i++) {
                setTimeout(createHeart, i * 100);
            }
        }

        // 4. Gemini AI Tip for the User:
        /* 
           PROMPT TO USE WITH GEMINI:

           They love [hobbies/style]. Write a 2-line poetic, futuristic birthday 
           wish I can put on a digital card."
           
           Then, replace the text inside <div id="special-message"> with Gemini's output!
        */
